
CREATE PROCEDURE SelectScoreDMX 
AS
SELECT 

a.SPlusCode,
a.Fullname,
a.Store,
a.Region,
COALESCE(b.Score,0)  as Score
FROM 

[dbo].[DearlerFSMUpload] as a
LEFT JOIN [dbo].[SplusActivityUpload] as b on a.[SPlusCode] = b.LoginID and year(b.AttempEnddate) = 2020 and month(b.AttempEnddate) = 12 and upper(a.SPlusCode) like 'CE.DMX%'

EXEC SelectScoreDMX
